using System;

namespace csharp_sample
{
    public class Program
    {
        static void Main(string[] args)
        {
            string[] line = Console.ReadLine().Split(" ");
            int a = int.Parse(line[0]);
            int b = int.Parse(line[1]);

            if (a * b % 2 == 0)
            {
                Console.WriteLine("Even");
            }
            else
            {
                Console.WriteLine("Odd");
            }
        }
    }
}
